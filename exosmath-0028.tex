\begin{exercice}\label{exosmath-0028}

    \begin{multicols}{2}
        \begin{enumerate}
            \item
                Simplifier
                \begin{equation}
                    \frac{ 14x+4 }{ 8 }
                \end{equation}
            \item
                Résoudre \( 4x-5=9\).
            \item
                Mettre au même dénominateur et effectuer la somme :
                \begin{equation}
                    \frac{1}{ x }+\frac{ 2 }{ 3a }.
                \end{equation}
            \item
                Mettre \( a\) en facteur dans l'expression \( a^3-\frac{ a }{2}\).
        \end{enumerate}
    \end{multicols}

\corrref{smath-0028}
\end{exercice}
