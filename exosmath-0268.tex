\begin{exercice}\label{exosmath-0268}

Donner les ensembles de définition des fonctions suivantes :
\begin{multicols}{2}
  \begin{enumerate}
\item $f(x) = \dfrac1{x+3}$
\item $g(x) = 2x^2-3x+1$
\item $h(x) = \dfrac{x-1}{3x-2}$
\item $k(x) = \dfrac3{x} - 4x^2 +2x -1$
\end{enumerate}
\end{multicols}

\end{exercice}
