\begin{exercice}\label{exosmath-0343}

    Parmi les expressions suivantes, laquelle est égale à \( 3x+2-(x+3)(3x+2)\) ?
    \begin{enumerate}
        \item
            \( -(3x+2)(x+2)\)
        \item
            \( (3x+2)(x-2)\)
        \item
            \( (3x+2)(-x+5)\).
    \end{enumerate}
    Même énoncé pour \( (7-x)(3x+10)+2x(x-7)\).
    \begin{enumerate}
        \item
            \( (7-x)(x+10)\)
        \item
            \( (7-x)(5+10x)\)
        \item
            \( (x-7)(5x+10)\)
    \end{enumerate}

\corrref{smath-0343}
\end{exercice}
