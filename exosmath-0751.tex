\begin{exercice}\label{exosmath-0751}

Sans les calculer, donner le signe des quotients suivants :
\begin{multicols}{2}
    \begin{enumerate}
        \item
            \( (-3)\div(-8) \)
        \item
            \( (+1)\div (-2)\)
        \item
            \( (-4)\div (-5)\)
        \item
            \( (-3.7)\div (+5.1)\)
    \end{enumerate}
\end{multicols}

\end{exercice}
