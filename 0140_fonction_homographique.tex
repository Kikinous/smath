\section{Définitions}
%+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++

\begin{definition}
    Une \defe{fonction homographique}{homographique} est une fonction de la forme
    \begin{equation}
        f(x)=\frac{ ax+b }{ cx+d }.
    \end{equation}
Son ensemble de définition est l'ensemble des \( x\in \eR\) tels que \( cx+d\neq 0\).
\end{definition}
