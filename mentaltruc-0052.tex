\begin{mental}
    Si 
    \begin{equation}
        55=\frac{ 127 }{ a },
    \end{equation}
    combien vaut \( a\) ?
    \begin{enumerate}
        \item
            \( 55\times 127\)
        \item
            \( 55\div 127\)
        \item
            \( 127\div 55\).
    \end{enumerate}
\end{mental}
