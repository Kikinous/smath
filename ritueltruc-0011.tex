\begin{rituel}
    Trouver des nombres \( a\) et \( b\) tels que
    \begin{subequations}
        \begin{numcases}{}
            a+b=37\\
            a-b=13
        \end{numcases}
    \end{subequations}
    en même temps.
\end{rituel}
