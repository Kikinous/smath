\begin{exercice}\label{exosmath-0504}

    Dresser le tableau de signe des fonctions suivantes :
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( f(x)=\frac{ 2x+3 }{ 2 }\)
            \item
                \( g(x)=-\frac{ 1+x }{ 7 }\)
            \item
                \( h(x)=\frac{ x }{ 3 }-9\)
            \item
                \( k(x)=4-x\)
            \item
                \( l(x)=-\frac{ x }{ 7 }+\frac{1}{ 7 }\).
        \end{enumerate}
    \end{multicols}

\corrref{smath-0504}
\end{exercice}
