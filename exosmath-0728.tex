    Entourer en vert le premier calcul à effectuer, et calculer :
    \begin{multicols}{2}

\item
$A = 11 + 18 - 2$
\item
$B = 17 - 9 - 2$
\item
$C = 3\times 8 + 2$
\item
$D = 27 - 18 + 2$
\item
$E = 7 + 3\times5$
\item
$F = 12 - 2\times 5$
        \end{enumerate}
    \end{multicols}
