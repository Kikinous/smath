    Donner la valeur des expressions suivantes si \( a=3\) :
    \begin{multicols}{3}
        \begin{enumerate}
            \item
                \( a+5\)
            \item
                \( 2\times a\)
            \item
                \( a+2\)
            \item
                \( a-1\)
            \item
                \( a\times 9\)
            \item
                \( a\times a\)
        \end{enumerate}
    \end{multicols}
