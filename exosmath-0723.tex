\begin{exercice}\label{exosmath-0723}

Compléter les égalités :
\begin{enumerate}
    \item
        \( -2+\ldots=7\)
    \item
        \( 4+(-\ldots)=-1\)
    \item
        \( -2+(-\ldots)=-5\)
    \item
        \( -2+\ldots=-7\)
\end{enumerate}

\corrref{smath-0723}
\end{exercice}
