\begin{exercice}\label{exosmath-0848}

    Entourer en vert le premier calcul à effectuer, et calculer :
    \begin{multicols}{2}
        \begin{enumerate}
\item
    \( A=4\times 8+3\)
\item
    \( B=(5+3)\times 2+10\)
\item
    \( \dfrac{ 9\times 4 }{ 18 }\)
\item
    \( \dfrac{ 12 }{ 4\times 7 }\)
\item
    \( \dfrac{ 3 }{ 2+1 }+4\)
\item
    \( 1-\dfrac{ 10 }{ 5 }\)
        \end{enumerate}
    \end{multicols}

\end{exercice}
