\begin{exercice}\label{exosmath-0513}

    Tracer dans un repère orthonormé le graphe des fonctions affines suivantes :
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( l(x)=8x\)
            \item
                \( g(x)=-x+3\)
            \item
                \( h(x)=5x-7\)
            \item
                \( k(x)=-7x-1\)
            \item
                \( f(x)=2x\)
            \item
                \( c(x)=-2\)
        \end{enumerate}
    \end{multicols}

\corrref{smath-0513}
\end{exercice}
