\begin{exercice}\label{exosmath-0739}

    Compléter les égalités suivantes :
    \begin{enumerate}
        \item
            \( 7\times 35+7\times 24=7\times(\ldots +\ldots)\)
        \item
            \( 7\times 35-7\times 24=7\times(\ldots - \ldots)\)
        \item
            \( 12\times 5+8\times 12 = \ldots\times (5+8) \)
        \item
            \( 3\times 1.4-3\times 0.8=(1.4\ldots 0.8)\ldots 3\)
    \end{enumerate}

\end{exercice}
