\begin{exercice}\label{exosmath-0274}

    Résoudre les équations suivantes.
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( \frac{1}{ x }=6\)
            \item
                \( \frac{1}{ x }=-\frac{ 1 }{2}\)
            \item
                \( \frac{1}{ x }=\frac{ 13 }{ 17 }\)
            \item
                \( 4+\frac{1}{ x }=-7\)
            \item
                \( 2=\frac{ 2 }{ 3x+7 }\)
        \end{enumerate}
    \end{multicols}

\end{exercice}
