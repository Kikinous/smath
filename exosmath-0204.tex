\begin{exercice}\label{exosmath-0204}

    \begin{enumerate}
        \item
            \( (3-x)(2x+5)\) vaut
            \begin{enumerate}
                \item
                    \( (x-3)(2x-5)\)
                \item
                    \( -(x-3)(2x+5)\)
            \end{enumerate}
        \item
            \( -(5x+1)(x-3)\) vaut
            \begin{enumerate}
                \item
                    \( (5x+1)(3-x)\)
                \item
                    \( (-5x+1)(x-3)\)
                \item
                    \( (-5x-1)(-x+3)\)
                \item
                    \( -(5x+1)(x+3)\)
            \end{enumerate}
    \end{enumerate}

\corrref{smath-0204}
\end{exercice}
