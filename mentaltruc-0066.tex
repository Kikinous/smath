\begin{mental}
    
Si \( x>5\), alors
\begin{enumerate}
    \item
        \( x+3<x+5\)
    \item
        \( \dfrac{ x }{ 2 }>10\)
    \item
        \( 2x>10\)
\end{enumerate}
\end{mental}
