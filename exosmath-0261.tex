                \( x\in\mathopen] 2 , 10 \mathclose[\)
            \item
                \( x\in\mathopen[ -4 , -1 [\)
            \item
                \( x\in\mathopen] -2 , 13 \mathclose]\)
        \end{enumerate}
    \end{multicols}
