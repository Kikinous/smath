\begin{exercice}\label{exo2smath-0104}

    Écrire sans les parenthèses :
    \begin{multicols}{2}
    \begin{enumerate}
        \item
            \( 18+(7+3)\)
        \item
            \( 18+(7-3)\)
        \item
            \( 100-(100+10)\)
        \item
            \( 12+(a+5)\)
        \item
            \( 12-(b+7)\)
        \item
            \( x-(4-x)\)
    \end{enumerate}
    \end{multicols}

\corrref{2smath-0104}
\end{exercice}
