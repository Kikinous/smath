\begin{exercice}[\cite{NRHooXFvgpp5}]\label{exo2smath-0054}

    Pour chacune des suites, donner les trois termes suivants :
    \begin{enumerate}
        \item
            \( 36\), \( 35\), \( 34\), \ldots
        \item
            $-36$ ; $-35$ ; $-34$ ; \ldots
        \item
            \( -34\), \( -35\), \( -36\)
\item
 $6$ ; $4$ ; $2$ ; \ldots
\item
 $-50$ ; $-40$ ; $-30$ ;\ldots
            \item
                \( 5.5\), \( 6\), \( 6.5\), \( 7\), \ldots
            \item
                \( -3.7\), \( -3.8\), \( -3.9\), \ldots
            \item
                \( -3.3\), \( -3.2\),  \( -3.1\), \ldots
    \end{enumerate}
