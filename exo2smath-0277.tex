\begin{exercice}\label{exo2smath-0277}

    Lesquelles des égalités suivantes sont correctes si \( x=3\) ?
    
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( 2\times x+3=9\)
            \item
                \( 2x-3=9\)
            \item
                \( 5\times x=0\)
            \item
                \( x-3=0\)
            \item
                \( x-7=-4\)
            \item
                \( -2x=6\)
        \end{enumerate}
    \end{multicols}

\end{exercice}
