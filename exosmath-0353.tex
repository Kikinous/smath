\begin{exercice}%[\cite{oklaEg}]
    \label{exosmath-0353}

    \begin{enumerate}
        \item
            Factoriser \( (x+1)\) dans l'expression \( 4(x+1)-(x+1)(x+3)\).
        \item
Résoudre l'inéquation
\begin{equation}
    \frac{ 4(x+1) }{ x+3 }\geq x+1
\end{equation}
en se servant éventuellement du premier point.
    \end{enumerate}
Attention aux valeurs interdites.

\corrref{smath-0353}
\end{exercice}
