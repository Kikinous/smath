\begin{exercice}\label{exosmath-0346}

    Résoudre les équations et inéquations suivantes.
    \begin{multicols}{2}
    \begin{enumerate}
        \item
            \begin{equation}
                -4x-2=\frac{ 10 }{ -6 }.
            \end{equation}
        \item
            \begin{equation}
                \frac{ 1 }{ 8x+3 }=-5
            \end{equation}
        \item
            \begin{equation}
                \frac{ 1 }{ x }\leq\frac{ 1 }{ 3 }
            \end{equation}
        \item
            \begin{equation}
                4\leq x^2\leq 16.
            \end{equation}
    \end{enumerate}
    \end{multicols}
    Donner les réponses sous forme de fractions simplifiées.

\end{exercice}
