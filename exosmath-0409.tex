\begin{exercice}\label{exosmath-0409}

    Dire si les fonctions suivantes sont affines (c'est à dire du type \( ax+b\)) et, le cas échéant, préciser \( a\) et \( b\).
    \begin{multicols}{3}
        \begin{enumerate}
            \item
                \( f(x)=3x+8\)
            \item
                \( g(x)=\frac{ 2 }{ x }-7\)
            \item
                \( h(x)=1-x\)
            \item
                \( k(x)=-\frac{ 1 }{2}x\).
        \end{enumerate}
    \end{multicols}

\corrref{smath-0409}
\end{exercice}
