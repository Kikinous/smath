\begin{exercice}\label{exosmath-0471}

    Effectuer les opérations et simplifier les fractions suivantes.
    \begin{multicols}{3}
        \begin{enumerate}
            \item
                \( \frac{ x }{ 2 }+\frac{ a }{ 3 }\)
            \item
                \( \frac{ 10x+2 }{ 4 }\)
            \item
                \( \frac{ x^2-16 }{ x-4 }\)
            \item
                \( \frac{1}{ x+a }-\frac{ 2 }{ (x-a) }\)
            \item
                \( \frac{ 4x+2a }{ 2 }-2x\)
            \item
                \( \frac{ 3x(x+2) }{ x(x-1) }+\frac{ 2x^2 }{ x(x-1) }\)
        \end{enumerate}
    \end{multicols}

\corrref{smath-0471}
\end{exercice}
