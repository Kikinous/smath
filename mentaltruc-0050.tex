\begin{mental}
    Résoudre les équations suivantes :
    \begin{enumerate}
        \item
            \begin{equation}
                2x=8
            \end{equation}
        \item
            \begin{equation}
                2=\frac{ 12 }{ x }
            \end{equation}
        \item
            \begin{equation}
                x+7=-10
            \end{equation}
    \end{enumerate}
\end{mental}
