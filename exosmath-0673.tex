\begin{exercice}%[\ldots/3]
    \label{exosmath-0673}

    Questions type «calcul mental».
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                Simplifier
                \begin{equation*}
                    \frac{ a^2+2a }{ 8a }
                \end{equation*}
            \item
                La fonction \( f(x)=-3x+7\) est-elle croissante ou décroissante ?
            \item
                Résoudre
                \begin{equation*}
                    \frac{1}{ x }=\frac{ 4 }{ 7 }.
                \end{equation*}
            \item
                Développer \( (x-4)^2\).
            \item
                Donner un encadrement de \( f(-3)\) 
                \begin{equation*}
                    \begin{array}[]{c|ccccccc}
                        x&-4&&0&&2&&7\\
                        \hline
                        &7&&&&4&&\\
                        f(x)&&\searrow&&\nearrow&&\searrow&\\
                        &&&3&&&&3\\
                    \end{array}
                \end{equation*}
        \end{enumerate}
    \end{multicols}

\corrref{smath-0673}
\end{exercice}
