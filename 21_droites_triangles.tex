\section{Médiatrice}
%+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++

\begin{definition}
    Si \( A\) et \( B\) sont de points, la \defe{médiatrice}{médiatrice} du segment \( [AB]\) est la droite perpendiculaire à \( (AB)\) et passant par le milieu de \( [AB]\).
\end{definition}

\section{Cercle circonscrit}
%+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++

\begin{definition}
    Le \defe{cercle circonscrit}{cercle!circonscrit} à un triangle est le cercle passant par les trois sommets de ce triangle.
\end{definition}
