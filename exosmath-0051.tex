\begin{exercice}\label{exosmath-0051}

Soit \( f(x)=x^2+6x+1\).
\begin{enumerate}
    \item
        Factoriser \( f(x)+8\).
    \item
        En déduire que \( f\) admet un minimum en \( x=-3\).
\end{enumerate}

\end{exercice}
