\begin{exercice}\label{exo2smath-0193}

    Calculer la valeur des expressions suivantes si \( t=12\).
    \begin{multicols}{3}
        \begin{enumerate}
            \item
                \( 2\times t\)
            \item
                \( 5t\)
            \item
                \( t+2\)
            \item
                \( t\div 4\)
            \item
                \( \dfrac{ 120 }{ t }\)
        \end{enumerate}
    \end{multicols}
