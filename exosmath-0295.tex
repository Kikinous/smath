\begin{exercice}\label{exosmath-0295}

    \begin{enumerate}
        \item
            Donner un exemple de fonction pour laquelle \( 4\) est l'image de \( 2\).
        \item
            Donner un exemple de fonction pour laquelle \( 2\) est un antécédent de \( 10\).
        \item
            Résoudre l'équation
            \begin{equation}
                3x+10=7.
            \end{equation}
            Simplifier au maximum la réponse.
    \end{enumerate}

\end{exercice}
