\begin{exercice}\label{exosmath-0158}

Donner les trois premiers termes des suites
\begin{multicols}{3}
    \begin{enumerate}
        \item
            \( u_n=n+1\)
        \item
            \( u_n=3n-1\)
        \item
            \( u_n=2^n\)
        \item
            \( u_n=1-n\)
        \item
            \( u_n=n^2\)
        \item
            \( u_n=\frac{ 2 }{ n }\).
    \end{enumerate}
\end{multicols}

\corrref{smath-0158}
\end{exercice}
