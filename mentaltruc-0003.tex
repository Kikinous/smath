    \begin{mental}
        Si \( 6\) paquets de bonbons coûtent \( 3\) euros et \( 5\) paquets en coûtent \( 2.5\),
        \begin{enumerate}
            \item
                combien coûtent \( 10\) paquets de bonbons ?
            \item
                combien coûtent \( 11\) paquets de bonbons ?
        \end{enumerate}
    \end{mental}
