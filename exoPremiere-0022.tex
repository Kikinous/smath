\begin{exercice}\label{exoPremiere-0022}

    \begin{multicols}{2}
    \begin{enumerate}
        \item
            Calculer
            \begin{equation}
                \frac{ x }{ 2 }\times\frac{ 4 }{ x^2 }.
            \end{equation}
        \item
            Factoriser
            \begin{equation}
                x^2-9
            \end{equation}
        \item
            Résoudre
            \begin{equation}
                \frac{ 3 }{ 4 }x+5=10
            \end{equation}
        \item
            Effectuer
            \begin{equation}
                -2(x+3-x^3).
            \end{equation}
    \end{enumerate}
    \end{multicols}
