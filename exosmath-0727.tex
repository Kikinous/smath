\begin{exercice}\label{exosmath-0727}

    Mettre sous forme de somme :
    \begin{enumerate}
        \item
            \( A=(-3)\times 2\)
        \item
            \( B=2\times (-3)\)
        \item
            \( C=(-5)\times 4\)
        \item
            \( D=1.5\times 8\)
        \item
            \( E=(-1.5)\times 8\)
    \end{enumerate}
    Effectuer ces opérations.

\corrref{smath-0727}
\end{exercice}
