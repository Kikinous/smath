\begin{exercice}[\cite{ZNloWAM}]\label{exosmath-0317}

Soit la fonction
\begin{equation}
    f(x)=\frac{ (x-\frac{1}{ 3 })(3x+2)-(9x^2-4) }{ (\frac{ 4 }{ 3 }-x) }.   
\end{equation}
\begin{enumerate}
    \item
        En utilisant un produit remarquable, comment peut-on récrire \( 9x^2-4\) ?
    \item
        Factoriser \( f(x)\).
    \item
        Résoudre \( f(x)=0\).
\end{enumerate}

\corrref{smath-0317}
\end{exercice}
