\begin{exercice}\label{exosmath-0173}

    Écrire une fonction qui implémente la fonction
    \begin{equation}
        f(x)=\begin{cases}
            x+1    &   \text{si \( x\geq 4\)}\\
            2x    &    \text{si \( x<4\)}
        \end{cases}.
    \end{equation}
    Faire calculer \( f(1)\), \( f(-1)\), \( f(30)\) et \( f(4)\).

\corrref{smath-0173}
\end{exercice}
