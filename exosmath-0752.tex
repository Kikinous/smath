\begin{exercice}\label{exosmath-0752}

    Parmi les expressions ci-dessous, lesquelles sont égales à \( \dfrac{ -2 }{ 3 }\) ?
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                 \( 2\div(-3)\)
             \item
                 \( -0.66\)
             \item
                 \( \dfrac{ 2 }{ -3 }\)
             \item
                 \( -\dfrac{ 2 }{ 3 }\)
             \item
                \( 2\div 3\)
        \end{enumerate}
    \end{multicols}

\corrref{smath-0752}
\end{exercice}
