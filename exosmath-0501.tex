\begin{exercice}\label{exosmath-0501}

    Dire si les fonctions suivantes sont croissantes ou décroissantes.
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( f(x)=4x-2\)
            \item
                \( g(x)=-12x-10\)
            \item
                \( h(x)=\frac{ 1 }{ 2 }x+12\)
            \item
                \( k(x)=\frac{ 2x-4 }{ 3 }\)
            \item
                \( l(x)=\frac{ 1-5x }{ 2 }\)
            \item
                \( r(x)=-\frac{ -x+3 }{ 5 }\)
            \item
                \( q(x)=-\frac{ 1-x }{ 7 }\)
        \end{enumerate}
    \end{multicols}

\corrref{smath-0501}
\end{exercice}
