\begin{exercice}\label{exosmath-0640}

    \begin{enumerate}
        \item
            Combien vaut \( f(4)\) si \( f(x)=x^2-2\) ?
        \item
            Combien vaut \( g(-1)\) si \( g(x)=3x-1\) ?
        \item
            Combien vaut \( h(10)\) si \( h(x)=\frac{1}{ x+2 }\) ?
    \end{enumerate}
