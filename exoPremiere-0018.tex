\begin{exercice}\label{exoPremiere-0018}

    Pour la fonction \( f(x)=1/(x^2+4x)\),
    \begin{enumerate}
        \item
            donner le domaine de définition,
        \item
            calculer si possible \( f(0)\) et \( f(-1)\),
        \item
            donner le ou les antécédents de \( -1/4\).
    \end{enumerate}

\end{exercice}
