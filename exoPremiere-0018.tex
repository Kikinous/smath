\begin{exercice}\label{exoPremiere-0018}

    Pour la fonction \( f(x)=\frac{\displaystyle 1}{\displaystyle x^2+4x }\),
    \begin{enumerate}
        \item
            calculer si possible \( f(-1)\) et \( f(0)\),
        \item
            donner l'ensemble de définition,
        \item
            donner le ou les antécédents de \( -1/4\).
    \end{enumerate}

\end{exercice}
