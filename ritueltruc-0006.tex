\begin{rituel}
    Parmi les expressions suivantes, dire lesquelles sont égales à
    \begin{equation}
        \frac{ 5\times 7 }{ 3\times 11 }.
    \end{equation}
    \begin{multicols}{2}
    \begin{enumerate}
        \item
            \begin{equation}
                5\times \frac{ 7 }{ 3\times 11 }
            \end{equation}
        \item
            \begin{equation}
                3\times \frac{ 5\times 7 }{ 11 }
            \end{equation}
        \item
            \begin{equation}
                \frac{ 5\times 7 }{ 11 }\div 3
            \end{equation}
        \item
            \begin{equation}
                \frac{ 5 }{ 33 }\times 7
            \end{equation}
    \end{enumerate}
    \end{multicols}
\end{rituel}
