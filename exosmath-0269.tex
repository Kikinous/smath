\item Résoudre : $(3+5x)^2 -(4x-7)(3+5x)=0$
  \smallskip
\item 
  \begin{enumerate}
  \item Montrer que l'équation \ $(7x+1)^2=(4-8x)^2$ \ est équivalente à
    l'équation \\[1ex] $(15x-3)(-x+5)=0$.
  \item Résoudre cette dernière équation et conclure.
  \end{enumerate}
  \smallskip
\item Résoudre : $x^2-1+(x-1)(4x+3)=0$
\end{enumerate}
