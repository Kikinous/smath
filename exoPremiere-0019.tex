\begin{exercice}\label{exoPremiere-0019}

    \begin{enumerate}
        \item
            Calculer
            \begin{equation}
                \frac{ 1 }{2}-\frac{1}{ 4 }.
            \end{equation}
        \item
            Mettre au même dénominateur
            \begin{equation}
                \frac{1}{ x }+\frac{ 5 }{ 4 }.
            \end{equation}
        \item
            Résoudre 
            \begin{equation}
                5x=4
            \end{equation}
            en laissant le résultat sous forme de fractions.
        \item
            Simplifier
            \begin{equation}
                \frac{ 5x }{ x^2 }
            \end{equation}
    \end{enumerate}

\corrref{Premiere-0019}
\end{exercice}
