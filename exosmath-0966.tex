\begin{exercice}\label{exosmath-0966}

    Recopier les expressions suivantes et calculer leurs valeurs lorsque \( x=5\).
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( x+5\)
            \item
                \( -x\)
            \item
                \( 4\times x\)
            \item
                \( -2x\)
            \item
                \( x\times x\)
        \end{enumerate}
    \end{multicols}

\corrref{smath-0966}
\end{exercice}
