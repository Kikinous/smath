\begin{exercice}\label{exoPremiere-0061}

    Développer (c'est à dire distribuer) l'expression
    \begin{equation}
        -8(x-\frac{ 1 }{2})(x+\frac{ 3 }{ 4 }).
    \end{equation}
    Résoudre ensuite l'inéquation
    \begin{equation}
        -8x^2-2x+3>0.
    \end{equation}

\end{exercice}
