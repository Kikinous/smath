\begin{exercice}[\ldots /2]\label{exosmath-0772}

    Parmi les expressions suivantes, lesquelles sont égales à \( -\dfrac{ 6 }{ 7 }\) ?
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( \dfrac{ 6 }{ -7 }\)
            \item
                \( -0.87\)
            \item
                \( -(6\div 7)\)
            \item
                \( \dfrac{ 7 }{ 6 }\).
        \end{enumerate}
    \end{multicols}

\corrref{smath-0772}
\end{exercice}
