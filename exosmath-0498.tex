\begin{exercice}\label{exosmath-0498}

    Résoudre les inéquations suivantes. Donner les solutions sous forme d'intervalle.
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( 2x<10\).
            \item
                \( 3x+5\leq 20\)
            \item
                \( -x+2\geq 0\)
            \item
                \( 10x-7>3x+2\).
        \end{enumerate}
    \end{multicols}

\corrref{smath-0498}
\end{exercice}
