\begin{exercice}[\cite{ZNloWAM}]\label{exosmath-0320}

    Soit la fonction
    \begin{equation}
        f(x)=x^2-\frac{ 10 }{ 7 }x-\frac{ 8 }{ 7 }.
    \end{equation}
    \begin{enumerate}
        \item
            Pour quelle valeur de \( a\) avons-nous \( f(x)=(x-2)(x+a)\) ?
        \item
            Résoudre l'équation \( f(x)=0\).
    \end{enumerate}

\corrref{smath-0320}
\end{exercice}
