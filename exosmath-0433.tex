\begin{exercice}\label{exosmath-0433}

    Résoudre les équations suivantes :
    \begin{multicols}{3}
        \begin{enumerate}
            \item
                \( 7(-x-9)=0\).
            \item
                \( x(3x+2)=0\).
            \item
                \( (x-1)(x+3)=0\).
            \item
                \( \frac{ x+2 }{ 5x }=0\).
            \item
                \( \frac{ 7x-3 }{ -3x+4 }=0\).
        \end{enumerate}
    \end{multicols}

\corrref{smath-0433}
\end{exercice}
