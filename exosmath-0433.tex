\begin{exercice}\label{exosmath-0433}

    Résoudre les équations et inéquations suivantes :
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( 7(-x-9)\leq 0\).
            \item
                \( x(3x+2)=0\).
            \item
            \( (x-1)(x+3)\geq 0\).
            \item
                \( (x+2)^2-49=0\)
            \item
                \( 5x(x+4)-(3x+2)(4+x)=0\).
            \item
                \( \frac{ 7x-3 }{ -3x+4 }<0\).
        \end{enumerate}
    \end{multicols}

\corrref{smath-0433}
\end{exercice}
