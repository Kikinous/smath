\begin{exercice}\label{exosmath-0969}

    \begin{enumerate}
        \item
            Laquelle de ces trois expressions est égale à \(  30\times (10-x)  \) ?
            \begin{enumerate}
                \item
                    \( 300-30x\)
                \item
                    \( 30-10x\)
                \item
                    \( 30+30x\)
            \end{enumerate}
        \item
            Développer : \( 3\times (x+2)\)
        \item
            Factoriser : \( 6\times a-3\times a^2\).
    \end{enumerate}

\end{exercice}
