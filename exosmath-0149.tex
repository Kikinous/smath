\begin{exercice}\label{exosmath-0149}

    Dire si les fonctions suivantes sont affines (\( ax+b\)) et, le cas échéant, préciser \( a\) et \( b\).
    \begin{multicols}{3}
        \begin{enumerate}
            \item
                \( f(x)=3x+8\)
            \item
                \( g(x)=x-1\)
            \item
                \( h(x)=\frac{ 2 }{ x }-7\)
            \item
                \( i(x)=1-x\)
            \item
                \( j(x)=x^2+x+1\)
            \item
                \( k(x)=-\frac{ 1 }{2}x\).
        \end{enumerate}
    \end{multicols}

\corrref{smath-0149}
\end{exercice}
