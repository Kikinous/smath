\begin{exercice}\label{exosmath-0642}

    Résoudre les équation suivantes :
    \begin{multicols}{2}
    \begin{enumerate}
        \item
            \( 2x=8\).
        \item
            \( 3x+5=35\).
        \item
            \( 7x+10=18\).
        \item
            \( (x-6)(x-12)=0\).
        \item
            \( (x+1)(x-5)=0\).
        \item
            \( 2x+4=-3x+19\)
    \end{enumerate}
    \end{multicols}

\end{exercice}
