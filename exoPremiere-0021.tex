\begin{exercice}\label{exoPremiere-0021}

    \begin{enumerate}
        \item
            Résoudre
            \begin{equation}
                10x+5=5.
            \end{equation}
        \item
            Calculer
            \begin{equation}
                \frac{ 3 }{ 4 }+\frac{ 5 }{ 8 }.
            \end{equation}
        \item
            Mettre au même dénominateur
            \begin{equation}
                \frac{ 1 }{ 2 }-\frac{ a }{ x }
            \end{equation}
        \item
            Effectuer
            \begin{equation}
                x^2(x^{-3}+2).
            \end{equation}
    \end{enumerate}

\end{exercice}
