\begin{exercice}[\ldots\ldots/4]\label{exo2smath-0088}

    Laquelle des expressions suivantes est égale à \( \dfrac{ 3 }{ 7 }\) ?
    \begin{multicols}{4}
        \begin{enumerate}
            \item
                \( \dfrac{ 21 }{ 49 }\)
            \item
                \( \dfrac{ 7 }{ 3 }\)
            \item
                \( \dfrac{ 3+2 }{ 7+2 }\)
            \item
    $0.42857142857143$
        \end{enumerate}
    \end{multicols}

    Laquelle des expressions suivantes n'est pas égale à \( \dfrac{ 3 }{ 7 }\) ?
    \begin{multicols}{4}
        \begin{enumerate}
            \item
                \( \dfrac{ 1.5 }{ 3.5 }\)
            \item
                \( \dfrac{ 3\times 124 }{ 7\times 124 }\)
            \item
                \( \dfrac{ 1 }{ 2 }+\dfrac{ 2 }{ 5 }\)
            \item
                \( \dfrac{ 5 }{ 7 }-\dfrac{ 2 }{ 7 }\)
        \end{enumerate}
    \end{multicols}

\corrref{2smath-0088}
\end{exercice}
