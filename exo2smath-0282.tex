\begin{exercice}\label{exo2smath-0282}

    Pour quelle valeur de \( x\) les égalités suivantes sont vraies ?
    \begin{multicols}{2}
        \begin{enumerate}
        \item
            \( x+1=3\)
        \item
            \( 2x=8\)
        \item
            \( x+3=5\)
        \item
            \( 3x+4=35\)
        \item
            \( x-5=10\)
        \item
            \( x+5=3\)
    \end{enumerate}
    \end{multicols}

\corrref{2smath-0282}
\end{exercice}
