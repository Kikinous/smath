\begin{exercice}\label{exoPremiere-0067}

    \begin{enumerate}
        \item
    Écrire une fonction qui prend un nombre \info{x} en argument et qui retourne \info{``oui''} si il est divisible par \( 7\) et \info{``non''} si il ne l'est pas.
    \item

        Écrire une fonction qui prend deux nombres \info{x} et \info{b} en argument et qui retourne \info{``oui''} si \info{x}  est divisible par \info{b} et \info{``non''} si il ne l'est pas.

    \end{enumerate}
