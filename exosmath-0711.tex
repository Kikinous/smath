\begin{exercice}\label{exosmath-0711}

    \begin{enumerate}
        \item
            
    Montrer que l'inéquation \( \frac{ x+1 }{ 3x+2 }\leq \frac{ x-2 }{ 3x+1 }\) est équivalente à l'inéquation \( \frac{ 12x-3 }{ (3x+2)(3x+1) }\leq 0\).
\item
    Résoudre cette inéquation.
    \end{enumerate}

\end{exercice}
