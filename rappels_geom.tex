\section{Triangles isométriques}
%+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++

Voir \cite{TqcjwY}.

\begin{propriete}       \label{PropRtqqxJ}
    Soient les triangles \( ABC\) et \( MNP\). Si
    \begin{enumerate}
        \item
            \( \hat A=\hat M\)
        \item
            \( AB=MN\)
        \item
            \( AC=MP\)
    \end{enumerate}
    alors les triangles sont isométriques.

    Si
    \begin{enumerate}
        \item
            \( AB=MN\)
        \item
            \( \hat A=\hat M\)
        \item
            \( \hat B=\hat N\)
    \end{enumerate}
    alors les triangles sont isométriques.
\end{propriete}
