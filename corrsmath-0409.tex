\begin{corrige}{smath-0409}

    \begin{enumerate}
        \item
            Elle est affine avec \( a=3\) et \( b=8\).
        \item
            Elle n'est pas affine parce qu'il y a un \( x\) dans un dénominateur.
        \item
            Elle est affine avec \( a=-1\) et \( b=1\).
        \item
            Elle est affine avec \( a=-\frac{ 1 }{2}\) et \( b=0\).
    \end{enumerate}

\end{corrige}
