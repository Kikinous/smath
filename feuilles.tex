\end{definition}

\begin{Aretenir}
    À propos de droites coplanaires :
    \begin{enumerate}
        \item
            Deux droites sécantes sont toujours coplanaires.
        \item
            Deux droites parallèles sont coplanaires.
    \end{enumerate}
\end{Aretenir}

%///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
