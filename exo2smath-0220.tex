\begin{exercice}[Programme de calcul\cite{NRHooXFvgpp5}]\label{exo2smath-0220}

    \begin{itemize}
        \item 
Choisir un nombre ;
\item
Lui retrancher $5$ ;
\item
Si le résultat est inférieur à \( -3\), lui ajouter \( 12\), sinon, lui ajouter \( -9\).
    \end{itemize}

    \begin{enumerate}
        \item
Appliquer ce programme à $6$ puis à \( -3\).
\item
 On obtient $15$ comme résultat. Quel est le nombre choisi au départ ?
    \end{enumerate}

\end{exercice}
