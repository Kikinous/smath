\begin{exercice}\label{exo2smath-0102}

    Relier chaque forme développée à sa forme factorisée :
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( 3x+6\)
            \item
                \( 12a+8b\)
            \item
                \( 6x+3\)
                \columnbreak
            \item
                \( 4(3a+2b)\)
            \item
                \( 3(x+2)\)
            \item
                \( 3(2x+1)\)
        \end{enumerate}
    \end{multicols}

\corrref{2smath-0102}
\end{exercice}
