\begin{exercice}\label{exosmath-0743}

    Calculer les produits suivants :
    \begin{multicols}{2}
    \begin{enumerate}
        \item
            \( 4\times 3\)
        \item
            \( 4\times (-3)\)
        \item
            \( 3\times (-4)\)
        \item
            \(  (-5)\times (-6)  \)
        \item
            \( -7\times 2\)
        \item
            \( 7\times (-2)\)
    \end{enumerate}
    \end{multicols}

\end{exercice}
