\begin{exercice}\label{exosmath-0514}

    \begin{enumerate}
        \item
            Soit \( f\colon x\mapsto 3x-1\). Calculer \( f(2)\), \( f(-1)\) et \( f(34)\).
        \item
            Soit \( g\colon x\mapsto -10x+12\). Calculer \( f(-3)\), \( f(\sqrt{2})\) et \( f(0)\).
    \end{enumerate}

\end{exercice}
