\begin{exercice}\label{exosmath-0144}

    \begin{enumerate}
        \item
            On sait que \( x\geq 0\). Comparer \( \frac{1}{ x+1 }\) et \( \frac{1}{ x+3 }\).
        \item
            On sait que \( x\leq 0\). Comparer \( \frac{1}{ x-\sqrt{5} }\) et \( \frac{1}{ x-2 }\).
    \end{enumerate}

\end{exercice}
