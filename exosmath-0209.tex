\begin{exercice}\label{exosmath-0209}

    Soit une fonction définie sur \( \mathopen[ -20 , 10 \mathclose]\) dont le tableau de variation est
\begin{equation*}
    \begin{array}[]{c|ccccc}
        x&-20&&-3&&10\\
        \hline
        &&&-6&&\\
        f(x)&&\nearrow&&\searrow&\\
        &-10&&&&-15
    \end{array}
\end{equation*}
Comparer \( f(5)\) et \( f(9)\).

\corrref{smath-0209}
\end{exercice}
