\begin{exercice}\label{exosmath-0047}

    Factoriser les expressions suivantes en utilisant des identités remarquables ou des mises en facteur.
    \begin{multicols}{3}
        \begin{enumerate}
            \item
                \( x^2+2x+1\)
            \item
                \( x^2-2x+1\)
            \item
                \( 4x^2+4x\)
            \item
                \( (x-2)^2-9\)
            \item
                \( x^2-9\)
            \item
                \( 4x^2+12x+9\)
            \item
                \( (x+1)^2-16\)
        \end{enumerate}
    \end{multicols}

\corrref{smath-0047}
\end{exercice}
