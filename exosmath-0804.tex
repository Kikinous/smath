\begin{exercice}\label{exosmath-0804}

    Donner les valeurs des expressions suivantes lorsque \( x=2\) :
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( 3x\)
            \item
                \( x+1\)
            \item
                \( -x\)
            \item
                \( 3x+9\)
            \item
                \( 2(x-3)+9\)
            \item
                \( x^2-4\)
        \end{enumerate}
    \end{multicols}

\corrref{smath-0804}
\end{exercice}
