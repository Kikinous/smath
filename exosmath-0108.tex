\begin{exercice}\label{exosmath-0108}

    Dire pour chacun des couples de vecteurs suivants si il sont colinéaires.
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( \vect{ u }=\begin{pmatrix}
                    2    \\ 
                    4   
                \end{pmatrix}\) et \( \vect{ v }=\begin{pmatrix}
                    6    \\ 
                    12    
                \end{pmatrix}\)
            \item
                \( \vect{ u }=\begin{pmatrix}
                    -1    \\ 
                    7    
                \end{pmatrix}\) et \( \vect{ v }=\begin{pmatrix}
                    1    \\ 
                    -7    
                \end{pmatrix}\)
            \item
                \( \vect{ u }=\begin{pmatrix}
                    7    \\ 
                    10    
                \end{pmatrix}\) et \( \vect{ v }=\begin{pmatrix}
                    1    \\ 
                    3    
                \end{pmatrix}\)
            \item
                \( \vect{ u }=\begin{pmatrix}
                    0    \\ 
                    4    
                \end{pmatrix}\) et \( \vect{ v }=\begin{pmatrix}
                    0    \\ 
                    \sqrt{2}    
                \end{pmatrix}\)
            \item
                \( \vect{ u }=\begin{pmatrix}
                    5    \\ 
                    0    
                \end{pmatrix}\) et \( \vect{ v }=\begin{pmatrix}
                    0    \\ 
                    1    
                \end{pmatrix}\)
            \item
                \( \vect{ u }=\begin{pmatrix}
                    \sqrt{5}+1    \\ 
                        1
                    \end{pmatrix}\) et \( \vect{ v }=\begin{pmatrix}
                        4    \\ 
                        \sqrt{5}-1    
                    \end{pmatrix}\).
        \end{enumerate}
    \end{multicols}

\corrref{smath-0108}
\end{exercice}
