\begin{exercice}\label{exoSeconde-0065}

    \begin{enumerate}
        \item
            Mettre \( 2\) en évidence dans l'expression \( 2x+4\).
        \item
            Simplifier
            \begin{equation}
                \frac{ 3x+6 }{ 3 }.
            \end{equation}
        \item
            Résoudre
            \begin{equation}
                \frac{ 6x }{ 7 }=3.
            \end{equation}
        \item
            Développer \( x^2(1+x)\).
    \end{enumerate}

\end{exercice}
