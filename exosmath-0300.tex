    \begin{equation*}
        \begin{array}[]{|c||c|c|c|c|c|c|}
            \hline
            \text{Point}&A(-2;4)&B(-1;1)&C(-0.5;0.25)&D(0.5;0.25)&E(1;1)&F(2;4)\\
            \hline\hline
            \text{Coefficient directeur}&-4&-2&-1&1&2&4\\
            \hline
        \end{array}
    \end{equation*}
