            \( E \) d'abscisse \( 5\) et \( F\) d'abscisse \( 2.5\)
        \item
            \( G \) d'abscisse \( 2\) et \( H\) d'abscisse \( -3\)
        \item
            \( K \) d'abscisse \( 2.4\) et \( L\) d'abscisse \( -10\)
        \item
            \( M \) d'abscisse \( -7\) et \( N\) d'abscisse \( -4.5\)
    \end{enumerate}
