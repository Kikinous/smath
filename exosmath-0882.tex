\begin{exercice}\label{exosmath-0882}

    Effectuer les produits suivants :
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( \dfrac{ 1 }{ 2 }\times \dfrac{ 3 }{ 4 }\)
            \item
                \( \dfrac{ 7 }{ 12 }\times 2\)
            \item
                \( \dfrac{ 7 }{ 12 }\times \dfrac{ 1 }{ 2 }\)
            \item
                \( \dfrac{ 8 }{ 4 }\times \dfrac{ 6 }{ 12 }\)
        \end{enumerate}
    \end{multicols}

\corrref{smath-0882}
\end{exercice}
