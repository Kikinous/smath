\begin{exercice}\label{exosmath-0306}

    Soit la suite définie par récurrence par
    \begin{subequations}
        \begin{numcases}{}
            u_0=3\\
            u_{n+1}=5u_n.
        \end{numcases}
    \end{subequations}
    \begin{enumerate}
        \item
            Déterminer la raison et le terme initial de cette suite. 
        \item
            Calculer \( u_1\), \( u_2\) et \( u_3\).
    \end{enumerate}

\corrref{smath-0306}
\end{exercice}
