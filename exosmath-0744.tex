\begin{exercice}\label{exosmath-0744}

    Calculer les expressions suivantes :
    \begin{enumerate}
        \item
            \( 3\times 4-21\)
        \item
            \( 3-4\times 2\)
        \item
            \( 7\times (-3)+7\times (+3)\)
        \item
            \( 7-13-17\)
        \item
            \( -3\times (2+9)+20\)
    \end{enumerate}

\end{exercice}
