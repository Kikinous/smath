\begin{exercice}\label{exo2smath-0097}

    Effectuer et simplifier les réponses :
    \begin{multicols}{4}
        \begin{enumerate}
            \item
                \( \dfrac{ 7 }{ 10 }+\dfrac{ 3 }{ 5 }\) 
            \item
                \( \dfrac{ 7 }{ 10 }-\dfrac{ 3 }{ 5 }\) 
            \item
                \( \dfrac{ 7 }{ 10 }\times\dfrac{ 3 }{ 5 }\) 
            \item
                \( \dfrac{ 7 }{ 10 }\div\dfrac{ 3 }{ 5 }\) 
        \end{enumerate}
    \end{multicols}
