\begin{exercice}\label{exoPremiere-0020}

    \begin{multicols}{2}
    \begin{enumerate}
        \item
            Résoudre
            \begin{equation}
                \frac{ x+2 }{ 3 }=4.
            \end{equation}
        \item
            Calculer
            \begin{equation}
                \frac{ 15 }{ 7 }\times \frac{ 7 }{ 5 }.
            \end{equation}
        \item
            Mettre au même dénominateur
            \begin{equation}
                \frac{1}{ x+2 }+\frac{1}{ x-1 }
            \end{equation}
        \item
            Simplifier 
            \begin{equation}
                \frac{ a+2a }{ a }.
            \end{equation}
    \end{enumerate}
    \end{multicols}

\corrref{Premiere-0020}
\end{exercice}
