\begin{exercice}[\cite{NRHooXFvgpp5}]\label{exosmath-0878}

    On donne \( a=\dfrac{ 1 }{ 6 }\), \( b=\dfrac{ 4 }{ 9 }\) et \( c=\dfrac{ 5 }{ 3 }\).
    \begin{enumerate}
        \item
            Calculer \( a\times b+a\times c\).
        \item
            Calculer \( a\times (b+c)\).
        \item
            Expliquer le résultat.
    \end{enumerate}

\end{exercice}
