\begin{exercice}[\cite{NRHooXFvgpp4}]\label{exo2smath-0108}

Réduire les expressions suivantes :
\begin{multicols}{2}
    \begin{enumerate}
        \item
            \( \dfrac{ 3x }{ 2 }+\dfrac{ x }{ 4 }\)
        \item
            \( \dfrac{ x }{ 6 }+\dfrac{ x-4 }{ 3 }\)
        \item
            \( 3+\dfrac{ x-1 }{ 5 }\)
        \item
            \( -5x-\dfrac{ 3x-2 }{ 4 }+3\)
    \end{enumerate}
\end{multicols}

\end{exercice}
