\begin{exercice}\label{exoSeconde-0060}

    Dessiner le graphe des fonctions suivantes :
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( f(x)=-x\)
            \item
                \( f(x)=3x\)
            \item
                \( f(x)=x+3\)
            \item
                \( f(x)=-2x+1\)
            \item
                \( f(x)=4\)
            \item
                \( f(x)=x\)
            \item
                \( f(x)=x^2\)
            \item
                \( f(x)=-x^2\)
        \end{enumerate}
    \end{multicols}

\corrref{Seconde-0060}
\end{exercice}
