\begin{mental}

    La base d'une pyramide est un carré. Donner
    \begin{enumerate}
        \item
            le nombre d'arêtes,
        \item
            le nombre faces,
        \item
            le nombre de sommets.
    \end{enumerate}

\end{mental}
