\begin{exercice}\label{exosmath-0637}

    Dire si les fonctions suivantes sont affines. Si oui, préciser le coefficient directeur et tracer dans un repère.
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( f(x)=3x+1\)
            \item
                \( g(x)=\frac{1}{ -x+1 }\)
            \item
                \( h(x)=-x+1\)
            \item
                \( k(x)=7x-7\)
            \item
                \( l(x)=3x^2-1\)
            \item
                \( m(x)=-2x+4\)
        \end{enumerate}
    \end{multicols}

\corrref{smath-0637}
\end{exercice}
