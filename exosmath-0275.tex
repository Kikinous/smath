\begin{exercice}\label{exosmath-0275}

    Représenter graphiquement les fonctions suivantes.
    \begin{multicols}{2}
        \begin{enumerate}
            \item
                \( f(x)=\frac{1}{ x }\)
            \item
                \( g(x)=\frac{ 2 }{ x }\)
            \item
                \( h(x)=-\frac{ 1 }{ x }\)
            \item
                \( k(x)=\frac{ 1 }{ x+4 }\)
            \item
                \( m(x)=\frac{1}{ x-3 }\)
        \end{enumerate}
    \end{multicols}

\end{exercice}
