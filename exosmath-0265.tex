\begin{exercice}[\cite{FGMWzeK}]\label{exosmath-0265}

    \begin{enumerate}
        \item
            Nous savons que \( x\geq 0\). Comparer \( \frac{1}{ x+4 }\) et \( \frac{1}{ x+1 }\).
        \item
            Nous savons que \( x\leq 0\). Comparer \( \frac{1}{ x-5 }\) et \( \frac{1}{ x-\sqrt{10} }\).
    \end{enumerate}
