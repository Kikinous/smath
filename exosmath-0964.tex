\begin{exercice}\label{exosmath-0964}

    Parmi les expressions suivantes, lesquelles sont égales à \( \dfrac{ 5 }{ 7 }\) ?
    \begin{multicols}{3}
        \begin{enumerate}
            \item
                $0.714$
            \item
                \( -\dfrac{ 5 }{ -7 }\)
            \item
                \( \dfrac{ 15 }{ 17 }\)
            \item
                \( \dfrac{ 15 }{ 21 }\)
            \item
                \( (-5)\div(-7)\)
        \end{enumerate}
    \end{multicols}

\corrref{smath-0964}
\end{exercice}
