\begin{exercice}\label{exosmath-0660}

    Résoudre dans \( \eR\) :
    \begin{multicols}{2}
        \begin{enumerate}
            \item
            \( x^2=4\)
        \item
            \( x^2+2=0\)
        \item
            \( x^2<36\)
        \item
            \( x^2\geq 16\).
        \end{enumerate}
    \end{multicols}

\end{exercice}
